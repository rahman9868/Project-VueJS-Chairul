new Vue({
	el: '#app',

 		 title: 'hello world!',
 	 	message: 'something'

  theTitle: function() {
    return this.title.toUpperCase();
  }
},

  lowercase: function(value){
    return value.toLowerCase(); 
  }
}  
});
